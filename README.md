# vue-touch-knob
### 安装
  ```
   npm i vue-touch-knob -S
  ```
### 使用
  ```
  <template>
    <div>
      <vue-knob></vue-knob>
    </div>
  </template>

  <script>
  import vueKnob from 'vue-touch-knob'
  export default {
    name: 'App',
    components: {vueKnob}
  }
</script>
  ```
